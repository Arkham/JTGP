        var add = function (a, b) {
          if (typeof b === 'undefined') {
            return function (x) {
              return a + x;
            }
          } else {
            return a + b;
          }
        };

        add(1, 3)
        // 4

        var addFour = add(4);

        addFour(99)
        // 103

        addFour(53)
